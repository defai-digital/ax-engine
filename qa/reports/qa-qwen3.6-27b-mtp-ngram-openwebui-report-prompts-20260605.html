function isValidEmail(email) {
  if (typeof email !== 'string') return false;
  
  const atIndex = email.indexOf('@');
  if (atIndex === -1) return false;
  
  // Check that there is exactly one @
  if (email.indexOf('@', atIndex + 1) !== -1) return false;
  
  // Check that there is at least one dot after @
  const afterAt = email.substring(atIndex + 1);
  if (afterAt.indexOf('.') === -1) return false;
  
  // Ensure the dot is not the last character (domain must have something after the dot)
  if (afterAt.endsWith('.')) return false;
  
  // Ensure there is something before @
  if (atIndex === 0) return false;
  
  return true;
}

sorting_algorithmPASSstop
- **Prerequisite and Setup**: Binary search operates exclusively on **sorted** arrays or lists, requiring the data to be ordered (ascending or descending) so that comparisons can effectively eliminate half of the remaining search space.
- **Divide and Conquer Strategy**: The algorithm compares the target value to the **middle element** of the current search range; if they match, the index is returned, otherwise, the search continues in the left half (if the target is smaller) or the right half (if the target is larger).
- **Iterative Reduction**: By repeatedly discarding the half of the array that cannot contain the target, the search space is reduced by 50% with each step, continuing until the target is found or the search range becomes empty (indicating the target is not present).
- **Time Complexity**: It achieves a highly efficient **O(log n)** time complexity, where *n* is the number of elements, making it significantly faster than linear search for large datasets, while requiring **O(1)** auxiliary space in its iterative implementation.
